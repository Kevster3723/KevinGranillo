/*
 * Kevin Granillo, Ivy Newman, Trin Whitehat, Nolan Leist
 * June 7, 2025
 *
 * Pick-up Place and Return: grabs a box, reads size and color,
 * finds the corresponding path, places it on a platform, then returns.
 *
 * Obstacle Avoidance: finds gaps in walls without crossing white boundaries,
 * stops at the green line at the end.
 */

#include <Servo.h>

#define gripperButton 25
#define Trig 10
#define Echo 11

int Lcolor, Rcolor, Mcolor;
int BoxColor, gSize;
Servo Gripper;
int actualPosition = 180;
int PWMSignal, distance = 0, duration = 0, val1;
int Counter = 0, GreenCount = 0;

enum State_e { Start, InSearch, LineFollow, Place, End, StopProgram };
enum State_e gState = Start;
enum gripperState { Pickup, Grab, Line };
enum gripperState gripperState = Pickup;

void setup() {
  Color_Sensor_setup();
  GripperColorSetup();
  MovementSetup();
  Gripper.attach(24);
  pinMode(gripperButton, INPUT_PULLUP);
  pinMode(Trig, OUTPUT);
  pinMode(Echo, INPUT);
  delay(5000); // Allow sensors to start
}

void loop() {
  switch(gState) {
    case Start:
      Ultrasonic();
      delay(10);
      switch(gripperState) {
        case Pickup:
          MovFWD(35);
          actualPosition = 150;
          val1 = map(actualPosition, 0, 300, 0, 180);
          Gripper.write(val1);
          delay(100);
          if(distance < 10) {
            Stop();
            gripperState = Grab;
            actualPosition = 170;
            val1 = map(actualPosition, 0, 300, 0, 180);
            Gripper.write(val1);
            delay(100);
          }
        break;
        case Grab:
          if(digitalRead(gripperButton) == HIGH) {
            actualPosition -= 2;
            val1 = map(actualPosition, 0, 300, 0, 180);
            Gripper.write(val1);
            delay(100);
          } else {
            gSize = (actualPosition < 122) ? 0 : 1; // 0=small, 1=big
            delay(500);
            GripperColorSensor();
            gripperState = Line;
          }
          if(actualPosition < 110) {
            actualPosition = 150;
            val1 = map(actualPosition, 0, 300, 0, 180);
            Gripper.write(val1); delay(100);
          }
        break;
        case Line:
          if(Lcolor==3 || Mcolor==3 || Rcolor==3) {
            MovFWD(35); delay(500);
            (BoxColor==1) ? RotateRt(35) : RotateLft(35);
            delay(1000); Stop();
            gState = InSearch;
          } else {
            ColorSensor();
            MovBKWD(30);
          }
        break;
      }
    break;
    case InSearch:
      ColorSensor(); LineSearch(3);
    break;
    case LineFollow:
      Ultrasonic(); delay(10); ColorSensor(); LineFollowing();
      if((Lcolor==3&&Mcolor==3)||(Mcolor==3&&Rcolor==3)||(Rcolor==3&&Lcolor==3))
        gState = Place;
    break;
    case Place:
      ColorSensor(); Ultrasonic(); LinePlace();
    break;
    case End:
      ColorSensor(); LineFollowing();
      if((Lcolor==3&&Mcolor==3)||(Mcolor==3&&Rcolor==3)||(Rcolor==3&&Lcolor==3)) GreenCount++;
      if(GreenCount==6) gState = StopProgram;
    break;
    case StopProgram: Stop(); break;
  }
}

void Ultrasonic() {
  digitalWrite(Trig, LOW); delayMicroseconds(2);
  digitalWrite(Trig, HIGH); delayMicroseconds(10);
  digitalWrite(Trig, LOW);
  duration = pulseIn(Echo, HIGH);
  distance = duration * 0.034 / 2;
}

void MovFWD(float speed) {
  float PWMval = 255*(speed/100);
  analogWrite(EnableLM1, PWMval); analogWrite(EnableRM1, PWMval);
  analogWrite(EnableLM2, PWMval); analogWrite(EnableRM2, PWMval);
  // All four wheels spin CW → forward motion
  digitalWrite(RRM_in1,HIGH); digitalWrite(RRM_in2,LOW);
  digitalWrite(FRM_in3,HIGH); digitalWrite(FRM_in4,LOW);
  digitalWrite(FLM_in1,LOW);  digitalWrite(FLM_in2,HIGH);
  digitalWrite(RLM_in3,LOW);  digitalWrite(RLM_in4,HIGH);
}

// TCS230/TCS3200 color sensors — Right, Middle, Left
#define RSOut 4  #define RS0 3  #define RS1 2  #define RS2 1  #define RS3 0
#define MSOut 14 #define MS0 15 #define MS1 16 #define MS2 17 #define MS3 18
#define LSOut 19 #define LS0 20 #define LS1 21 #define LS2 22 #define LS3 23

// Calibrated min/max frequency values per sensor per channel
int LredMin=81,  LredMax=528,  RredMin=90,  RredMax=732,  MredMin=83,  MredMax=584;
int LgreenMin=79,LgreenMax=560,RgreenMin=89,RgreenMax=805,MgreenMin=80,MgreenMax=628;
int LblueMin=68, LblueMax=499, RblueMin=73, RblueMax=680, MblueMin=72, MblueMax=579;

void Color_Sensor_setup() {
  // Set S0–S3 as OUTPUT, sensorOut as INPUT on each sensor
  // Set frequency scaling to 20%: S0=HIGH, S1=LOW
}

void ColorSensor() {
  // For each sensor (R, L, M):
  //   1. Select color filter via S2/S3 pin combo
  //   2. Read pulse width with pulseIn()
  //   3. map() frequency → 0–255 color value
  //   4. Classify: Black / White / Red / Green / Blue / Other

  // Example for Right sensor, RED channel:
  digitalWrite(RS2,LOW); digitalWrite(RS3,LOW);
  RredFreq  = pulseIn(RSOut, LOW);
  RredColor = map(RredFreq, RredMin, RredMax, 255, 0);

  // Classification logic (same pattern for all three sensors):
  // 0=Black  1=Red  2=Blue  3=Green  4=White  5=Other
  if (RredColor<130 && RgreenColor<130 && RblueColor<130)   Rcolor = 0; // Black
  else if (RredColor>250 && RgreenColor>250 && RblueColor>250) Rcolor = 4; // White
  else if (RredColor > RgreenColor && RredColor > RblueColor)  Rcolor = 1; // Red
  else if (RgreenColor > RredColor && RgreenColor > RblueColor) Rcolor = 3; // Green
  else if (RblueColor > RgreenColor && RblueColor > RredColor)  Rcolor = 2; // Blue
  else                                                            Rcolor = 5; // Other
  // Repeat for L and M sensors...
}
// Dedicated color sensor mounted in the gripper to identify box color
#define GSOut 9
#define GS0 8  #define GS1 7
#define GS2 6  #define GS3 5

// Calibrated frequency ranges
int GredMin=47,  GredMax=450;
int GgreenMin=48,GgreenMax=468;
int GblueMin=38, GblueMax=357;

void GripperColorSetup() {
  pinMode(GS0,OUTPUT); pinMode(GS1,OUTPUT);
  pinMode(GS2,OUTPUT); pinMode(GS3,OUTPUT);
  pinMode(GSOut,INPUT);
  digitalWrite(GS0,HIGH); digitalWrite(GS1,LOW); // 20% scaling
}

void GripperColorSensor() {
  // Uses AvgColor() helper for more stable readings
  digitalWrite(GS2,LOW);  digitalWrite(GS3,LOW);
  GredFreq   = AvgColor(GSOut);
  GredColor  = map(GredFreq,  GredMin,  GredMax,  255, 0);

  digitalWrite(GS2,HIGH); digitalWrite(GS3,HIGH);
  GgreenFreq  = AvgColor(GSOut);
  GgreenColor = map(GgreenFreq,GgreenMin,GgreenMax,255, 0);

  digitalWrite(GS2,LOW);  digitalWrite(GS3,HIGH);
  GblueFreq  = AvgColor(GSOut);
  GblueColor = map(GblueFreq, GblueMin, GblueMax, 255, 0);

  // Classify and store in BoxColor (1=Red, 2=Blue, etc.)
  if      (GredColor<100 && GgreenColor<100 && GblueColor<100) BoxColor = 0;
  else if (GredColor>200 && GgreenColor>200 && GblueColor>200) BoxColor = 4;
  else if (GredColor > GgreenColor && GredColor > GblueColor)   BoxColor = 1; // Red → rotate right
  else if (GgreenColor > GredColor+20 && GgreenColor > GblueColor) BoxColor = 3;
  else if (GblueColor > GgreenColor && GblueColor > GredColor)  BoxColor = 2; // Blue → rotate left
  else                                                            BoxColor = 5;
}
extern int gSize;
int LineCounter = 0, TurnCount = 0;

// Follows the line matching BoxColor using three ground-facing color sensors
void LineFollowing() {
  if      (Lcolor == BoxColor) RotateLft(30);
  else if (Mcolor == BoxColor) MovFWD(30);
  else if (Rcolor == BoxColor) RotateRt(30);
  else                         MovFWD(30);
}

// Rotates in place searching for a specific color line
void LineSearch(int Color) {
  if      (Lcolor == Color) RotateLft(30);
  else if (Mcolor == Color) MovFWD(30);
  else if (Rcolor == Color) RotateRt(30);
  else if (Rcolor == BoxColor && BoxColor == 1) { // Found red line
    delay(500); RotateRt(35); delay(1100);
    gState = LineFollow;
  }
  else if (Lcolor == BoxColor && BoxColor == 2) { // Found blue line
    delay(500); RotateLft(35); delay(1100);
    gState = LineFollow;
  }
  else MovFWD(35);
}

// Navigates to the drop platform and releases the box
void LinePlace() {
  if (Lcolor==BoxColor && Mcolor==BoxColor && Rcolor==BoxColor && TurnCount==0) {
    MovFWD(35); delay(400);
    (gSize==1) ? RotateLft(35) : RotateRt(35); // Size determines drop side
    delay(1200); TurnCount++;
  } else if (distance < 5) {
    Stop(); delay(500);
    // Open gripper to release box
    actualPosition = 170;
    val1 = map(actualPosition, 0, 300, 0, 180);
    Gripper.write(val1); delay(100);
    // Back up, realign, return to start path
    MovBKWD(35); delay(500);
    RotateLft(35); delay(2400);
    MovFWD(35); delay(1200);
    gState = End;
  } else {
    LineFollowing(); // Keep following line to platform
  }
}
/*
 * Obstacle Avoidance — separate task from payload transport.
 * The robot navigates a walled arena, finds gaps using the
 * ultrasonic sensor, avoids crossing white boundary lines,
 * and stops when it reaches the green finish line.
 */

void ObstacleSetup() {
  // Shares the same ultrasonic (Trig/Echo) and color sensors
  // as the main task — no additional hardware needed
}

void ObstacleAvoidance() {
  ColorSensor();   // Read boundary and finish lines
  Ultrasonic();    // Read distance to walls/obstacles

  if (Lcolor==3 || Mcolor==3 || Rcolor==3) {
    Stop(); // Green line detected — task complete
    gState = StopProgram;
    return;
  }

  if (Lcolor==4 || Rcolor==4) {
    // White boundary detected — steer away
    (Lcolor==4) ? RotateRt(35) : RotateLft(35);
    delay(300);
    return;
  }

  if (distance < 15) {
    // Obstacle ahead — scan for open gap
    Stop(); delay(200);
    StrafeLft(35); delay(400);   // Try left gap
    Ultrasonic();
    if (distance < 15) {
      StrafeRt(35); delay(800);  // Try right gap
    }
  } else {
    MovFWD(40); // Clear path — advance
  }
}
